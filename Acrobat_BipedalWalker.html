1. Environment factories and seeding

Defines helper functions that create environments for training evaluation and video recording. Each worker receives a distinct deterministic seed so that different processes explore different trajectories while remaining reproducible. Evaluation environments use a fixed seed for consistent comparison across runs. Video environments write recordings directly into the directory for a specific run so that all artifacts for that run stay in one place.

def set_env_seed(env: gym.Env, seed: int):
    """
    Seed the environment RNG and local process RNGs.
    """
    obs, info = env.reset(seed=seed)
    if hasattr(env.action_space, "seed"):
        env.action_space.seed(seed)
    np.random.seed(seed)
    random.seed(seed)
    return obs, info


def make_env(env_id: str, worker_id: int = 0, base_seed: int = SEED) -> gym.Env:
    """
    Build a fresh env instance for a specific worker (actor-learner process).
    """
    env = gym.make(env_id)
    _ = set_env_seed(env, worker_seed(worker_id, base_seed))
    return env


def make_eval_env(env_id: str, base_seed: int = SEED) -> gym.Env:
    """
    Evaluation environment: deterministic seeding, no video wrapper.
    """
    env = gym.make(env_id)
    _ = set_env_seed(env, base_seed)
    return env

def moving_average(x, window: int = 20):
    """
    Compute a simple moving average over a one dimensional array.
    """
    x = np.asarray(x, dtype=np.float32)
    if x.size == 0:
        return x
    if window <= 1:
        return x

    if x.size < window:
        return np.full_like(x, np.nan, dtype=np.float32)

    ma = np.convolve(x, np.ones(window, dtype=np.float32) / window, mode="valid")
    pad = np.full(window - 1, np.nan, dtype=np.float32)
    return np.concatenate([pad, ma])


def plot_rewards(
    rewards,
    run_dir: str,
    filename: str,
    title: str,
    ma_window: int = 20,
):
    """
    Plot episode rewards with a moving average and save the figure into run_dir.
    """
    rewards = np.asarray(rewards, dtype=np.float32)
    ma = moving_average(rewards, window=ma_window)

    plt.figure(figsize=(8, 4.5))
    plt.plot(rewards, label="Episode reward", alpha=0.4)
    plt.plot(ma, label=f"MA({ma_window})")
    plt.xlabel("Episode")

    out_path = os.path.join(run_dir, filename)
    plt.savefig(out_path)
    plt.show()

3. Actor critic networks

Defines actor critic networks that share a common feedforward torso and attach different output heads for discrete and continuous control. Subsection 3.1 builds the network for discrete action spaces such as Acrobot and subsection 3.2 will later define the variant for continuous actions used in Bipedal Walker.

3.1 Discrete actor critic network

Here, defines an actor critic network for discrete action spaces for Acrobot. A two layer multilayer perceptron maps the observation vector to a latent representation then a policy head outputs logits over the available actions and a value head predicts the state value. Orthogonal initialization is applied to the linear layers with a smaller gain on the policy head for more stable behaviour at the start of training and unit gain on the value head for value regression. A simple configuration dataclass stores observation dimension action count and hidden sizes so that the same architecture can be recreated for different runs. A helper constructs a categorical policy distribution from logits and a greedy action method selects the action with highest logit for deterministic evaluation.

from dataclasses import dataclass
from torch.distributions import Categorical


@dataclass
class DiscreteModelConfig:
    obs_dim: int
    n_actions: int
    hidden_sizes: tuple = (256, 256)


class ActorCriticDiscrete(nn.Module):
    """
    Actor critic network for discrete action spaces.
    """

    def __init__(self, obs_dim: int, n_actions: int, hidden_sizes=(256, 256)):
        super().__init__()
        h1, h2 = hidden_sizes

        self.torso = nn.Sequential(
            nn.Linear(obs_dim, h1),
            nn.ReLU(),
            nn.Linear(h1, h2),
            nn.ReLU(),
        )
        self.policy_head = nn.Linear(h2, n_actions)
        self.value_head = nn.Linear(h2, 1)

        # Orthogonal initialization
        for m in self.modules():
            if isinstance(m, nn.Linear) and m not in (self.policy_head, self.value_head):
                nn.init.orthogonal_(m.weight, gain=nn.init.calculate_gain("relu"))
                nn.init.zeros_(m.bias)
        nn.init.orthogonal_(self.policy_head.weight, gain=0.01)
        nn.init.zeros_(self.policy_head.bias)
        nn.init.orthogonal_(self.value_head.weight, gain=1.0)
        nn.init.zeros_(self.value_head.bias)

    def forward(self, x: torch.Tensor):
        x = x.float()
        z = self.torso(x)
        logits = self.policy_head(z)
        value = self.value_head(z).squeeze(-1)
        return logits, value

    @staticmethod
    def dist_from_logits(logits: torch.Tensor) -> Categorical:
        return Categorical(logits=logits)

    @torch.no_grad()
    def greedy_action(self, obs_np: np.ndarray) -> int:
        self.eval()
        device = next(self.parameters()).device
        x = torch.tensor(obs_np, dtype=torch.float32, device=device).unsqueeze(0)
        logits, _ = self(x)
        return int(torch.argmax(logits, dim=1).item())


def build_discrete_model_from_config(cfg: DiscreteModelConfig) -> ActorCriticDiscrete:
    return ActorCriticDiscrete(cfg.obs_dim, cfg.n_actions, cfg.hidden_sizes)

3.2 Continuous actor critic network

This cell defines an actor critic network for continuous action spaces for Bipedal Walker. As in the discrete case a two layer multilayer perceptron maps the observation vector to a latent representation. A Gaussian policy head outputs the mean action vector and a trainable log standard deviation parameter while a value head predicts the state value. The same orthogonal initialization scheme is used with a smaller gain on the policy head and unit gain on the value head to stabilise optimisation. A configuration dataclass records the observation dimension action dimension and hidden sizes so that the same architecture can be reconstructed for different runs. Helper methods construct a Normal policy distribution from the mean and standard deviation and a greedy action method returns the mean action which is later clipped to the valid action range by the environment wrapper.

In[]:
from dataclasses import dataclass
from torch.distributions import Normal


@dataclass
class ContinuousModelConfig:
    obs_dim: int
    act_dim: int
    hidden_sizes: tuple = (256, 256)


class ActorCriticContinuous(nn.Module):
    """
    Actor critic network for continuous action spaces with a Gaussian policy.
    """

    def __init__(self, obs_dim: int, act_dim: int, hidden_sizes=(256, 256)):
        super().__init__()
        h1, h2 = hidden_sizes

        self.torso = nn.Sequential(
            nn.Linear(obs_dim, h1),
            nn.ReLU(),
            nn.Linear(h1, h2),
            nn.ReLU(),
        )
        self.mu_head = nn.Linear(h2, act_dim)
        self.value_head = nn.Linear(h2, 1)

        # Log standard deviation parameter per action dimension
        self.log_std = nn.Parameter(torch.zeros(act_dim))

        # Orthogonal initialization
        for m in self.modules():
            if isinstance(m, nn.Linear) and m not in (self.mu_head, self.value_head):
                nn.init.orthogonal_(m.weight, gain=nn.init.calculate_gain("relu"))
                nn.init.zeros_(m.bias)
        nn.init.orthogonal_(self.mu_head.weight, gain=0.01)
        nn.init.zeros_(self.mu_head.bias)
        nn.init.orthogonal_(self.value_head.weight, gain=1.0)
        nn.init.zeros_(self.value_head.bias)

    def forward(self, x: torch.Tensor):
        x = x.float()
        z = self.torso(x)
        mu = self.mu_head(z)
        value = self.value_head(z).squeeze(-1)
        std = torch.exp(self.log_std).expand_as(mu)
        return mu, std, value

    def dist_from_params(self, mu: torch.Tensor, std: torch.Tensor) -> Normal:
        return Normal(loc=mu, scale=std)

    @torch.no_grad()
    def greedy_action(self, obs_np: np.ndarray) -> np.ndarray:
        self.eval()
        device = next(self.parameters()).device
        x = torch.tensor(obs_np, dtype=torch.float32, device=device).unsqueeze(0)
        mu, std, _ = self(x)
        action = mu.squeeze(0).cpu().numpy()
        return action


def build_continuous_model_from_config(cfg: ContinuousModelConfig) -> ActorCriticContinuous:
    return ActorCriticContinuous(cfg.obs_dim, cfg.act_dim, cfg.hidden_sizes)

4. A2C loss and update helper

Defines a helper that performs one Advantage Actor Critic (A2C) update from a batch of rollout data. The function takes a model an optimizer states actions bootstrapped returns and advantages and computes the policy loss the value loss and an entropy bonus. For discrete control it uses a categorical policy distribution built from logits and for continuous control it uses a Normal policy distribution built from the mean and standard deviation. Advantages can be normalised to reduce variance. The combined loss trades off return maximisation value fit and entropy regularisation. Gradients are clipped to a maximum norm for stability then the optimizer step is applied. This helper is shared by Acrobot and Bipedal Walker and the caller only needs to specify whether the update is discrete or continuous.

In[]:
from typing import Dict, Literal

def a2c_update(
    model: nn.Module,
    optimizer: torch.optim.Optimizer,
    s: torch.Tensor,          # (T, obs_dim)
    a: torch.Tensor,          # (T,) for discrete or (T, act_dim) for continuous
    G: torch.Tensor,          # (T,) bootstrapped returns
    A: torch.Tensor,          # (T,) raw advantages
    *,
    control_type: Literal["discrete", "continuous"],
    normalize_adv: bool = True,
    entropy_coef: float = 0.01,
    value_coef: float = 0.5,
    max_grad_norm: float = 0.5,
) -> Dict[str, float]:
    """
    Perform one A2C update from a rollout segment. Returns scalars for logging.
    """
    model.train()

    # normalize advantages
    if normalize_adv:
        A = (A - A.mean()) / (A.std() + 1e-8)

    # Forward pass through the appropriate actor critic
    if control_type == "discrete":
        logits, values = model(s)                # ActorCriticDiscrete
        dist = model.dist_from_logits(logits)
        log_prob = dist.log_prob(a)              # (T,)
        entropy = dist.entropy().mean()
    elif control_type == "continuous":
        mu, std, values = model(s)               # ActorCriticContinuous
        dist = model.dist_from_params(mu, std)
        # Sum log prob and entropy over action dimensions
        log_prob = dist.log_prob(a).sum(dim=-1)  # (T,)
        entropy = dist.entropy().sum(dim=-1).mean()
    else:
        raise ValueError(f"Unknown control_type {control_type}")

    # Policy and value losses
    policy_loss = -(log_prob * A).mean()
    value_loss = 0.5 * (G - values).pow(2).mean()

    # Combined loss with entropy bonus
    loss = policy_loss + value_coef * value_loss - entropy_coef * entropy

    # Backward and gradient clipping
    optimizer.zero_grad()
    loss.backward()
    grad_norm = torch.nn.utils.clip_grad_norm_(model.parameters(), max_grad_norm)
    optimizer.step()

    return {
        "loss": float(loss.item()),
        "policy_loss": float(policy_loss.item()),
        "value_loss": float(value_loss.item()),
        "entropy": float(entropy.item()),
        "grad_norm": float(grad_norm if torch.is_tensor(grad_norm) else grad_norm),
    }

5. Returns and advantage computation

Defines a helper function that turns a sequence of rewards done flags and predicted state values into bootstrapped returns and advantages. The computation runs backwards in time applies the discount factor until a terminal step and uses a final bootstrap value when the rollout is truncated by the time horizon. The result is a pair of tensors for returns and advantages that can be passed directly into the A2C update helper. The same logic applies to Acrobot and Bipedal Walker so this function is shared across environments.

In[]:
def compute_returns_and_advantages(
    rewards: np.ndarray,
    dones: np.ndarray,
    values: torch.Tensor,
    last_value: float,
    gamma: float,
    device: torch.device,
) -> tuple[torch.Tensor, torch.Tensor]:
    """
    Compute bootstrapped returns and advantages for a single rollout.
    """
    T = len(rewards)
    returns = np.zeros(T, dtype=np.float32)
    g = float(last_value)

    for t in reversed(range(T)):
        g = rewards[t] + gamma * g * (1.0 - float(dones[t]))
        returns[t] = g

    G = torch.tensor(returns, dtype=torch.float32, device=device)
    A = G - values.detach()
    return G, A

def evaluate_greedy(
    env_id: str,
    model: nn.Module,
    control_type: Literal["discrete", "continuous"],
    n_episodes: int,
    run_dir: str | None = None,
    max_steps: int | None = None,
) -> List[float]:

    Run greedy evaluation episodes and return total rewards.
    """

    device = next(model.parameters()).device

    std_ret = float(np.std(returns)) if len(returns) > 0 else float("nan")
    print(f"Greedy evaluation mean {mean_ret:.2f}  std {std_ret:.2f}")

    return returns

8. A2C training loop

This cell defines a training loop for the Advantage Actor Critic (A2C) algorithm that works for any single environment at a time. The function creates an environment runs rollout segments of fixed length collects states actions rewards and done flags and uses the actor critic model to predict values. After each segment it computes bootstrapped returns and advantages then applies the A2C update helper defined earlier. Episode returns are tracked in a list so that the total reward per episode can be plotted later with the shared plotting function. The control type argument switches between discrete and continuous policies so that the same loop can train Acrobot and Bipedal Walker. The function returns the trained model the list of episode returns and a small log dictionary with training metadata.

In[]:
from typing import Literal, Dict, Any, List


def train_a2c_single_env(
    env_id: str,
    model: nn.Module,
    control_type: Literal["discrete", "continuous"],
    run_dir: str,
    total_env_steps: int = 200_000,
    rollout_len: int = 20,
    gamma: float = 0.99,
    entropy_coef: float = 0.01,
    value_coef: float = 0.5,
    max_grad_norm: float = 0.5,
    lr: float = 7e-4,
    log_every: int = 10_000,
) -> tuple[nn.Module, List[float], Dict[str, Any]]:
    """
    Train A2C on a single environment and return model, episode returns, and logs.
    """
    os.makedirs(run_dir, exist_ok=True)

    optimizer = optim.Adam(model.parameters(), lr=lr)
    env = make_env(env_id, worker_id=0, base_seed=SEED)

    episode_returns: List[float] = []
    current_return = 0.0
    steps_total = 0
    iteration = 0
    last_log_steps = 0
    wall_start = time.time()

    obs, info = env.reset()
    done = False
    truncated = False

    while steps_total < total_env_steps:
        iteration += 1

        # Rollout buffers for this segment
        states = []
        actions = []
        rewards = []
        dones = []
        values = []

        for t in range(rollout_len):
            # Store state
            states.append(np.array(obs, copy=True))

            # Forward pass and action sampling
            obs_tensor = torch.tensor(obs, dtype=torch.float32, device=device).unsqueeze(0)

            if control_type == "discrete":
                logits, value = model(obs_tensor)
                dist = model.dist_from_logits(logits)
                action_tensor = dist.sample()
                action = int(action_tensor.item())
            elif control_type == "continuous":
                mu, std, value = model(obs_tensor)
                dist = model.dist_from_params(mu, std)
                action_tensor = dist.sample()
                action = action_tensor.squeeze(0).cpu().numpy()
            else:
                raise ValueError(f"Unknown control_type {control_type}")

            values.append(float(value.squeeze(0).item()))
            actions.append(action)

            # Environment step
            obs, reward, done, truncated, info = env.step(action)
            rewards.append(float(reward))
            dones.append(bool(done))

            current_return += float(reward)
            steps_total += 1

            # Episode boundary
            if done or truncated:
                episode_returns.append(current_return)
                current_return = 0.0
                obs, info = env.reset()
                done = False
                truncated = False

            if steps_total >= total_env_steps:
                break

        # Bootstrap value for last state if not terminal
        if done or truncated:
            last_value = 0.0
        else:
            obs_tensor = torch.tensor(obs, dtype=torch.float32, device=device).unsqueeze(0)
            with torch.no_grad():
                if control_type == "discrete":
                    _, v_last = model(obs_tensor)
                else:
                    _, _, v_last = model(obs_tensor)
            last_value = float(v_last.squeeze(0).item())

        # Prepare tensors and update
        states_tensor = torch.tensor(np.asarray(states), dtype=torch.float32, device=device)
        if control_type == "discrete":
            actions_tensor = torch.tensor(actions, dtype=torch.long, device=device)
        else:
            actions_tensor = torch.tensor(np.asarray(actions), dtype=torch.float32, device=device)

        values_tensor = torch.tensor(np.asarray(values), dtype=torch.float32, device=device)

        G, A = compute_returns_and_advantages(
            rewards=np.asarray(rewards, dtype=np.float32),
            dones=np.asarray(dones, dtype=np.bool_),
            values=values_tensor,
            last_value=last_value,
            gamma=gamma,
            device=device,
        )

        stats = a2c_update(
            model=model,
            optimizer=optimizer,
            s=states_tensor,
            a=actions_tensor,
            G=G,
            A=A,
            control_type=control_type,
            normalize_adv=True,
            entropy_coef=entropy_coef,
            value_coef=value_coef,
            max_grad_norm=max_grad_norm,
        )

        # Logging
        if iteration == 1 or (steps_total - last_log_steps) >= log_every or steps_total >= total_env_steps:
            if episode_returns:
                avg10 = float(np.mean(episode_returns[-10:]))
            else:
                avg10 = float("nan")
            print(
                f"[A2C] it={iteration:>5} steps={steps_total:>8} avg10={avg10:7.2f} "
                f"loss={stats['loss']:.3f} pg={stats['policy_loss']:.3f} "
                f"vf={stats['value_loss']:.3f} H={stats['entropy']:.3f} gn={stats['grad_norm']:.3f}",
                flush=True,
            )
            last_log_steps = steps_total

    env.close()
    wall_time = time.time() - wall_start
    if episode_returns:
        avg10 = float(np.mean(episode_returns[-10:]))
    else:
        avg10 = float("nan")
