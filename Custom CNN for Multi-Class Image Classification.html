import torch
import torch.nn as nn
import torch.nn.functional as F

class MyCNN(nn.Module):
    def __init__(self):
        super(MyCNN, self).__init__()
        #Convolutional Layers to extract features
        self.conv_layers = nn.Sequential(
            # 1st Conv layer
            nn.Conv2d(in_channels=3, out_channels=32, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2), # Downsample: 28x28  14x14

            #2nd Conv layer
            nn.Conv2d(in_channels=32, out_channels=64, kernel_size=3, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2), # Downsample: 14x14  7x7

            # 3rd Conv layer removed after testing  accuracy gains were minimal relative to compute/memory cost.

            # # 3rd Conv layer
            # nn.Conv2d(in_channels=64, out_channels=128, kernel_size=3, padding=1),
            # nn.ReLU(),

            # Dropout
            nn.Dropout(0.25)
        )
        #Fully Connected Layers to classify
        self.fc_layers = nn.Sequential(
            nn.Flatten(),
            nn.Linear(64 * 7 * 7, 128),
            nn.ReLU(),
            nn.Linear(128, 36)
        )

    def forward(self, x):
        x = self.conv_layers(x) # Feature extraction
        x = self.fc_layers(x)   # Classification
        return x

def train_model(
    model,
    train_loader,
    val_loader,
    criterion,
    optimizer,
    device,
    num_epochs=10,
    model_name=None
):
    model.to(device) # Move the model to GPU or CPU
    # Initialize values
    train_losses, val_losses = [], []
    train_accuracies, val_accuracies = [], []

    for epoch in range(num_epochs):
        # Set model for training (enables dropout)
        model.train()
        total_loss, correct, total = 0.0, 0, 0

        # Training loop
        for images, labels in train_loader:
            images, labels = images.to(device), labels.to(device)

            # Forward pass and loss computation
            optimizer.zero_grad()
            outputs = model(images)
            loss = criterion(outputs, labels)
            loss.backward()
            optimizer.step()

            # Track values
            total_loss += loss.item()
            _, predicted = torch.max(outputs, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()

        # Compute training metrices for each epcoh
        train_loss = total_loss / len(train_loader)
        train_acc = correct / total
        train_losses.append(train_loss)
        train_accuracies.append(train_acc)

        # set model to evaluation moode (disable dropout / no grad)
        model.eval()
        val_loss, correct_val, total_val = 0.0, 0, 0

        # Validation  loop
        with torch.no_grad():
            for images, labels in val_loader:
                images, labels = images.to(device), labels.to(device)
                outputs = model(images)
                loss = criterion(outputs, labels)
                val_loss += loss.item()

                _, predicted = torch.max(outputs, 1)
                total_val += labels.size(0)
                correct_val += (predicted == labels).sum().item()
        # Compute validation metrices for each epoch
        val_loss /= len(val_loader)
        val_acc = correct_val / total_val
        val_losses.append(val_loss)
        val_accuracies.append(val_acc)

        print(f"Epoch [{epoch+1}/{num_epochs}] | "
              f"Train Loss: {train_loss:.4f}, Train Acc: {train_acc:.4f} | "
              f"Val Loss: {val_loss:.4f}, Val Acc: {val_acc:.4f}")

    if model_name:
        torch.save(model.state_dict(), f"{model_name}_model.pt")
        print(f"Model weights are being saved")

    return model, train_losses, val_losses, train_accuracies, val_accuracies

import pickle
import os

def evaluate_model(model, test_loader, criterion, device,
                   train_acc=None, train_loss=None, val_acc=None, val_loss=None,
                   title="Final Evaluation Summary",
                   save_results_as=None):
    # Set model to evaluation  mode (disables dropout)
    model.eval()
    # Initialize tracking variables
    test_loss = 0.0
    correct, total = 0, 0
    all_preds, all_labels = [], []

    # Disable gradient computation for efficiency
    with torch.no_grad():
        for images, labels in test_loader:
            # Move inputs to device (GPU or CPU)
            images, labels = images.to(device), labels.to(device)

            # Forward pass and loss computation
            outputs = model(images)
            loss = criterion(outputs, labels)
            test_loss += loss.item()

            # Compute number of correct predictions
            _, predicted = torch.max(outputs, 1)
            correct += (predicted == labels).sum().item()
            total += labels.size(0)

            #Store predictions and true labels for further analysis
            all_preds.extend(predicted.cpu().numpy())
            all_labels.extend(labels.cpu().numpy())

    # Final averaged test loss and accuracy
    test_loss /= len(test_loader)
    test_acc = correct / total

    # Print summary
    print(f"\n{title}")
    if train_acc is not None:
        print(f"Train Accuracy:    {train_acc[-1]:.4f}")
        print(f"Train Loss:        {train_loss[-1]:.4f}")
    if val_acc is not None:
        print(f"Val Accuracy:      {val_acc[-1]:.4f}")
        print(f"Val Loss:          {val_loss[-1]:.4f}")
    print(f"Test Accuracy:     {test_acc:.4f}")
    print(f"Test Loss:         {test_loss:.4f}")

    # Bundle everything for future analysis
    results_dict = {
        "train_acc": train_acc,
        "train_loss": train_loss,
        "val_acc": val_acc,
        "val_loss": val_loss,
        "test_acc": test_acc,
        "test_loss": test_loss,
        "all_preds": all_preds,
        "all_labels": all_labels
    }
    if save_results_as:
        with open(f"{save_results_as}_results.pkl", "wb") as f:
          pickle.dump(results_dict, f)
        print("Results are being saved.")

    return results_dict

Why These Mistakes Happen?

  • Visual ambiguity:Handwritten or stylized characters often blur the distinction between 'I' and '1', or 'O'and '0', especially with slight variations in tilt, loop shape, or thickness.

  • Feature overlap: Convolutional features may not fully separate such pairs without more training examples or sharper augmentation.

  • Class imbalance is not the issue here, rather, these are semantic/visual overlaps that challenge even human readers.
